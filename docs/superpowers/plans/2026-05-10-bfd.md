# BFD — Berlin Food Stories Delivery Check Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a local web tool where a user enters a Berlin address and sees which Berlin Food Stories–recommended restaurants can deliver there via Wolt, Lieferando, or Uber Eats, with deep links to the corresponding listing on each platform.

**Architecture:** FastAPI backend serves a single-page Vue 3 UI from `/`, plus `/api/*` routes. A weekly BFS scrape populates SQLite. On each address check, the request is geocoded and three platform adapters run in parallel; results are matched against the BFS list by coordinate proximity with a name tie-breaker. A 24-hour SQLite cache wraps each `(address, platform)` lookup.

**Tech Stack:** Python 3.12, uv (package manager), FastAPI, httpx, BeautifulSoup4, rapidfuzz, geopy, Playwright (Chromium), pytest, pytest-asyncio, respx (httpx mock).

**Reference spec:** [docs/superpowers/specs/2026-05-10-bfd-design.md](../specs/2026-05-10-bfd-design.md)

```python
import sqlite3
from contextlib import contextmanager
from pathlib import Path
from typing import Iterator

SCHEMA = """
CREATE TABLE IF NOT EXISTS bfs_places (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    slug        TEXT UNIQUE NOT NULL,
    name        TEXT NOT NULL,
    lat         REAL NOT NULL,
    lng         REAL NOT NULL,
    address     TEXT,
    bfs_url     TEXT,
    image_url   TEXT,
    categories  TEXT,
    last_seen   TIMESTAMP NOT NULL,
    deleted_at  TIMESTAMP
);

CREATE TABLE IF NOT EXISTS geocode_cache (
    address_norm TEXT PRIMARY KEY,
    lat          REAL NOT NULL,
    lng          REAL NOT NULL,
    postcode     TEXT,
    fetched_at   TIMESTAMP NOT NULL
);

CREATE TABLE IF NOT EXISTS delivery_cache (
    address_norm TEXT NOT NULL,
    platform     TEXT NOT NULL,
    payload      TEXT NOT NULL,
    status       TEXT NOT NULL,
    error        TEXT,
    fetched_at   TIMESTAMP NOT NULL,
    PRIMARY KEY (address_norm, platform)
);
"""


def init_db(path: Path) -> None:
    path.parent.mkdir(parents=True, exist_ok=True)
    with sqlite3.connect(path) as conn:
        conn.executescript(SCHEMA)


@contextmanager
def connect(path: Path) -> Iterator[sqlite3.Connection]:
    conn = sqlite3.connect(path)
    conn.row_factory = sqlite3.Row
    try:
        yield conn
        conn.commit()
    finally:
        conn.close()
```

```python
import re
import unicodedata

from rapidfuzz import fuzz


_UMLAUT_MAP = str.maketrans({
    "ä": "ae", "ö": "oe", "ü": "ue", "ß": "ss",
    "Ä": "ae", "Ö": "oe", "Ü": "ue",
})


def address(raw: str) -> str:
    """Normalise an address: lowercase, collapse whitespace, trim."""
    return re.sub(r"\s+", " ", raw.strip().lower())


def name(raw: str) -> str:
    """Normalise a restaurant name for fuzzy matching."""
    s = raw.lower().translate(_UMLAUT_MAP)
    # Strip remaining accents
    s = "".join(
        c for c in unicodedata.normalize("NFD", s) if not unicodedata.combining(c)
    )
    # Drop apostrophes and punctuation, collapse spaces
    s = re.sub(r"[^a-z0-9 ]", " ", s)
    return re.sub(r"\s+", " ", s).strip()


def name_similarity(a: str, b: str) -> float:
    """Return [0.0, 1.0] similarity between two names."""
    return fuzz.token_set_ratio(name(a), name(b)) / 100.0
```

```python
import math
from typing import Iterable, Protocol, TypeVar

from .config import MATCH_MAX_DISTANCE_M, MATCH_MIN_NAME_SIMILARITY
from .normalize import name_similarity


class HasCoords(Protocol):
    name: str
    lat: float
    lng: float


T = TypeVar("T", bound=HasCoords)
B = TypeVar("B", bound=HasCoords)


def haversine(lat1: float, lng1: float, lat2: float, lng2: float) -> float:
    """Great-circle distance in metres."""
    r = 6_371_000  # Earth radius in metres
    p1, p2 = math.radians(lat1), math.radians(lat2)
    dp = math.radians(lat2 - lat1)
    dl = math.radians(lng2 - lng1)
    a = math.sin(dp / 2) ** 2 + math.cos(p1) * math.cos(p2) * math.sin(dl / 2) ** 2
    return 2 * r * math.asin(math.sqrt(a))


def match(
    bfs_places: Iterable[B],
    platform_results: Iterable[T],
    max_distance_m: float = MATCH_MAX_DISTANCE_M,
    min_name_sim: float = MATCH_MIN_NAME_SIMILARITY,
) -> dict[int | str, T]:
    """Map each BFS place's `id` to its best platform match, if any.

    A platform restaurant matches a BFS place when:
    - it is within `max_distance_m` (great-circle) of the BFS coordinates, AND
    - among the close candidates, the one with the highest name similarity
      passes `min_name_sim`.
    """
    results: dict[int | str, T] = {}
    plat_list = list(platform_results)
    for bfs in bfs_places:
        nearby = [
            r for r in plat_list
            if haversine(bfs.lat, bfs.lng, r.lat, r.lng) <= max_distance_m
        ]
        if not nearby:
            continue
        best = max(nearby, key=lambda r: name_similarity(bfs.name, r.name))
        if name_similarity(bfs.name, best.name) >= min_name_sim:
            results[bfs.id] = best
    return results
```

```python
from dataclasses import dataclass

from bs4 import BeautifulSoup


@dataclass(frozen=True)
class BFSPlace:
    slug: str
    name: str
    lat: float
    lng: float
    address: str
    bfs_url: str
    image_url: str


def parse_places(places_html: str) -> list[BFSPlace]:
    """Parse the `placesList` HTML chunk from BFS map.json."""
    soup = BeautifulSoup(places_html, "html.parser")
    out: list[BFSPlace] = []
    for li in soup.select("li.places__list__place"):
        slug = li.get("data-name") or ""
        lat = float(li["data-lat"])
        lng = float(li["data-lon"])
        address = li.get("data-address", "")

        anchor = li.select_one("a[href]")
        bfs_url = anchor["href"] if anchor else ""

        h3 = li.select_one("h3")
        display_name = h3.get_text(strip=True) if h3 else slug

        img = li.select_one("img[src]")
        image_url = img["src"] if img else ""

        out.append(BFSPlace(
            slug=slug,
            name=display_name or slug,
            lat=lat,
            lng=lng,
            address=address,
            bfs_url=bfs_url,
            image_url=image_url,
        ))
    return out
```

```python
import datetime as dt
import sqlite3
from pathlib import Path

import httpx

from .config import (
    BFS_MAP_JSON_URL, BFS_MAP_PAGE_URL, DB_PATH,
    DEFAULT_USER_AGENT, HTTP_TIMEOUT_SEC,
)
from . import db as _db


async def fetch_places() -> list[BFSPlace]:
    """Fetch BFS map.json and parse it."""
    headers = {
        "User-Agent": DEFAULT_USER_AGENT,
        "Referer": BFS_MAP_PAGE_URL,
        "Accept": "application/json,*/*",
    }
    async with httpx.AsyncClient(timeout=HTTP_TIMEOUT_SEC) as client:
        resp = await client.get(BFS_MAP_JSON_URL, headers=headers)
        resp.raise_for_status()
        data = resp.json()
    return parse_places(data["placesList"])


def persist(places: list[BFSPlace], db_path: Path = DB_PATH) -> dict:
    """Upsert places, soft-delete those no longer present."""
    _db.init_db(db_path)
    now = dt.datetime.utcnow().isoformat()
    seen_slugs = {p.slug for p in places}

    added = updated = 0
    with _db.connect(db_path) as conn:
        existing = {
            row["slug"]: row["id"]
            for row in conn.execute("SELECT id, slug FROM bfs_places")
        }
        for p in places:
            if p.slug in existing:
                conn.execute(
                    """
                    UPDATE bfs_places
                       SET name=?, lat=?, lng=?, address=?, bfs_url=?, image_url=?,
                           last_seen=?, deleted_at=NULL
                     WHERE slug=?
                    """,
                    (p.name, p.lat, p.lng, p.address, p.bfs_url, p.image_url, now, p.slug),
                )
                updated += 1
            else:
                conn.execute(
                    """
                    INSERT INTO bfs_places(slug, name, lat, lng, address, bfs_url, image_url, last_seen)
                    VALUES (?, ?, ?, ?, ?, ?, ?, ?)
                    """,

            "UPDATE bfs_places SET deleted_at=? "

            "UPDATE bfs_places SET deleted_at=? WHERE deleted_at IS NULL", (now,)

        "updated": updated,
        "removed": removed,
```

```python
import datetime as dt
import json
from dataclasses import dataclass
from pathlib import Path
from typing import Optional

import httpx

from . import db as _db
from . import normalize
from .config import (
    DB_PATH, GEOCODE_CACHE_TTL_DAYS, HTTP_TIMEOUT_SEC,
    NOMINATIM_URL, NOMINATIM_USER_AGENT,
)


class AddressNotFound(Exception):
    pass


@dataclass(frozen=True)
class GeocodeResult:
    lat: float
    lng: float
    postcode: Optional[str]


async def geocode(address: str, db_path: Path = DB_PATH) -> GeocodeResult:
    addr_norm = normalize.address(address)
    cached = _read_cache(addr_norm, db_path)
    if cached:
        return cached

    params = {
        "q": address,
        "format": "jsonv2",
        "addressdetails": "1",
        "countrycodes": "de",
        "limit": "1",
    }
    headers = {"User-Agent": NOMINATIM_USER_AGENT}

    async with httpx.AsyncClient(timeout=HTTP_TIMEOUT_SEC) as client:
        resp = await client.get(NOMINATIM_URL, params=params, headers=headers)
        resp.raise_for_status()
        results = resp.json()

    if not results:
        raise AddressNotFound(f"Could not geocode: {address!r}")

    top = results[0]
    result = GeocodeResult(
        lat=float(top["lat"]),
        lng=float(top["lon"]),
        postcode=(top.get("address") or {}).get("postcode"),
    )
    _write_cache(addr_norm, result, db_path)
    return result


def _read_cache(addr_norm: str, db_path: Path) -> Optional[GeocodeResult]:
    cutoff = (
        dt.datetime.utcnow() - dt.timedelta(days=GEOCODE_CACHE_TTL_DAYS)
    ).isoformat()
    with _db.connect(db_path) as conn:
        row = conn.execute(
            "SELECT lat, lng, postcode FROM geocode_cache "
            "WHERE address_norm=? AND fetched_at>=?",
            (addr_norm, cutoff),
        ).fetchone()
    if row is None:
        return None
    return GeocodeResult(lat=row["lat"], lng=row["lng"], postcode=row["postcode"])


def _write_cache(addr_norm: str, r: GeocodeResult, db_path: Path) -> None:
    now = dt.datetime.utcnow().isoformat()
    with _db.connect(db_path) as conn:
        conn.execute(
            "INSERT OR REPLACE INTO geocode_cache(address_norm, lat, lng, postcode, fetched_at) "
            "VALUES (?,?,?,?,?)",
            (addr_norm, r.lat, r.lng, r.postcode, now),
        )
```

```python
import httpx

from ..config import DEFAULT_USER_AGENT, HTTP_TIMEOUT_SEC
from .base import DeliveryAdapter, DeliveryRestaurant


WOLT_API_URL = "https://restaurant-api.wolt.com/v1/pages/restaurants"


class WoltAdapter:
    name = "wolt"

    async def list_deliverable(
        self,
        lat: float,
        lng: float,
        address: str,
        postcode: str | None = None,
    ) -> list[DeliveryRestaurant]:
        params = {"lat": str(lat), "lon": str(lng)}
        headers = {
            "User-Agent": DEFAULT_USER_AGENT,
            "Accept": "application/json",
        }
        async with httpx.AsyncClient(timeout=HTTP_TIMEOUT_SEC) as client:
            resp = await client.get(WOLT_API_URL, params=params, headers=headers)
            resp.raise_for_status()
            data = resp.json()
        return list(_extract_restaurants(data))


def _extract_restaurants(data: dict):
    """Walk the Wolt API response and yield unique DeliveryRestaurants.

    The Wolt API groups restaurants into 'sections', each containing 'items'.
    The same restaurant can appear in multiple sections — dedupe by venue id.
    """
    seen: set[str] = set()
    for section in data.get("sections", []):
        for item in section.get("items", []):
            venue = item.get("venue") or {}
            venue_id = venue.get("id") or item.get("track_id") or ""
            if not venue_id or venue_id in seen:
                continue
            seen.add(venue_id)

            slug = venue.get("slug") or item.get("link", {}).get("target", "")
            coords = venue.get("location", {}).get("coordinates", {}) or {}
            lat_v = coords.get("lat") or item.get("venue", {}).get("location", {}).get("lat")
            lng_v = coords.get("lon") or item.get("venue", {}).get("location", {}).get("lon")
            if lat_v is None or lng_v is None:
                # Some Wolt items lack coordinates (campaigns, ads); skip.
                continue

            yield DeliveryRestaurant(
                platform="wolt",
                id=str(venue_id),
                name=venue.get("name") or item.get("title", ""),
                lat=float(lat_v),
                lng=float(lng_v),
                url=f"https://wolt.com/en/deu/berlin/restaurant/{slug}" if slug else "",
                address=venue.get("address", ""),
            )
```

```python
from fastapi import FastAPI, HTTPException

STATIC_DIR = Path(__file__).parent / "static"


@app.get("/api/health")
async def health() -> dict:
    return {"status": "ok"}


@app.get("/api/check")
async def check(address: str) -> dict[str, Any]:
    try:
        loc = await geocode.geocode(address)
    except geocode.AddressNotFound as e:
        raise HTTPException(status_code=404, detail=str(e))

    bfs_places = bfs.load_active()
    platforms: dict[str, Any] = {}
    matches_per_platform: dict[str, dict] = {}

    for adapter in ADAPTERS:
        try:
            results = await adapter.list_deliverable(
                lat=loc.lat, lng=loc.lng, address=address, postcode=loc.postcode,
            )
            matched = matcher.match(_as_match_records(bfs_places), results)
            matches_per_platform[adapter.name] = matched
            platforms[adapter.name] = {"status": "ok", "matched": len(matched), "from_cache": False}
        except Exception as e:
            platforms[adapter.name] = {"status": "error", "error": str(e)}

    by_slug = {p.slug: p for p in bfs_places}
    places_out: list[dict] = []
    for p in bfs_places:
        delivery: list[dict] = []
        for plat_name, matches in matches_per_platform.items():
            r = matches.get(p.slug)
            if r is not None:
                delivery.append({"platform": plat_name, "url": r.url})
        places_out.append({
            "bfs_name": p.name,
            "bfs_url": p.bfs_url,
            "image": p.image_url,
            "address": p.address,
            "delivery": delivery,
        })

    return {
        "address": address,
        "coords": {"lat": loc.lat, "lng": loc.lng},
        "checked_at": dt.datetime.utcnow().isoformat() + "Z",
        "platforms": platforms,
        "places": places_out,
    }


def _as_match_records(bfs_places: list[bfs.BFSPlace]):
    """Wrap BFSPlace so each has the `.id` attribute the matcher expects.

    BFSPlace's natural id is `slug` — expose it through a cheap proxy without
    changing the dataclass.

    class _R:
        __slots__ = ("id", "name", "lat", "lng")
        def __init__(self, p: bfs.BFSPlace):
            self.id = p.slug
            self.name = p.name
            self.lat = p.lat
            self.lng = p.lng
    return [_R(p) for p in bfs_places]
```

```python
from .base import DeliveryRestaurant

class LieferandoAdapter:
    name = "lieferando"

    async def list_deliverable(
        self,
        lat: float,
        lng: float,
        address: str,
        postcode: str | None = None,
    ) -> list[DeliveryRestaurant]:
        if not postcode:
            raise ValueError("Lieferando requires a postcode")

        params = {
            "postalCode": postcode,
            "latitude": str(lat),
            "longitude": str(lng),
            "country": "de",
            "isAccurate": "true",
        }
        headers = {
            "User-Agent": DEFAULT_USER_AGENT,
            "X-Country-Code": "de",
            "Accept": "application/json",
        }
        async with httpx.AsyncClient(timeout=HTTP_TIMEOUT_SEC) as client:
            resp = await client.get(LIEFERANDO_API_URL, params=params, headers=headers)
            resp.raise_for_status()
            data = resp.json()
        return list(_extract(data))


def _extract(data: dict):
    """Yield DeliveryRestaurant from a Takeaway/JustEat API response.

    Real response shape varies; the wrapper handles the common path
    `data.restaurants` (Takeaway) and `restaurants` (older JustEat).
    Adjust against the captured fixture if needed.

    candidates = (
        data.get("restaurants")
        or (data.get("data") or {}).get("restaurants")
        or []
    )

    for r in candidates:

        slug = r.get("primarySlug") or r.get("slug") or ""

            continue
```

```python
import asyncio
from typing import Any

from playwright.async_api import async_playwright, Response

from ..config import HTTP_TIMEOUT_SEC
from .base import AdapterUnavailable, DeliveryRestaurant


UBEREATS_HOME = "https://www.ubereats.com/de"
FEED_ENDPOINT_FRAGMENT = "getFeedV1"  # part of the request URL we intercept


class UberEatsAdapter:
    name = "ubereats"

    async def list_deliverable(
        self,
        lat: float,
        lng: float,
        address: str,
        postcode: str | None = None,
    ) -> list[DeliveryRestaurant]:
        feed_payload: dict[str, Any] | None = None
        feed_event = asyncio.Event()

        async with async_playwright() as p:
            browser = await p.chromium.launch(headless=True)
            try:
                context = await browser.new_context(
                    locale="de-DE",
                    geolocation={"latitude": lat, "longitude": lng},
                    permissions=["geolocation"],
                )
                page = await context.new_page()

                async def on_response(resp: Response):
                    nonlocal feed_payload
                    if FEED_ENDPOINT_FRAGMENT in resp.url:
                        try:
                            feed_payload = await resp.json()
                            feed_event.set()
                        except Exception:
                            pass

                page.on("response", on_response)
                await page.goto(UBEREATS_HOME, wait_until="domcontentloaded")

                # Type the address into the autocomplete and pick the first suggestion.
                # Selectors are best-effort; UE's UI changes frequently.
                try:
                    await page.fill("input[placeholder*='Adresse'], input[name='address']", address, timeout=10_000)
                    await page.keyboard.press("ArrowDown")
                    await page.keyboard.press("Enter")
                except Exception as e:
                    raise AdapterUnavailable(f"address input not found: {e}") from e

                try:
                    await asyncio.wait_for(feed_event.wait(), timeout=HTTP_TIMEOUT_SEC)
                except asyncio.TimeoutError as e:
                    raise AdapterUnavailable("getFeedV1 not observed within timeout") from e
            finally:
                await browser.close()

        if feed_payload is None:
            raise AdapterUnavailable("no feed response captured")
        return _parse_feed(feed_payload)


def _parse_feed(payload: dict) -> list[DeliveryRestaurant]:
    """Walk the Uber Eats feed JSON and yield restaurants.

    The feed is a list of items of various types (stores, banners, etc.).
    We only keep items with a store object that has coordinates.
    """
    out: list[DeliveryRestaurant] = []
    items = (
        payload.get("data", {}).get("feedItems")
        or payload.get("feedItems")
        or []
    )
    seen: set[str] = set()
    for item in items:
        store = item.get("store") or item.get("storeV2") or {}
        store_id = str(store.get("uuid") or store.get("storeUuid") or "")
        if not store_id or store_id in seen:
            continue
        loc = store.get("location") or {}
        lat = loc.get("latitude")
        lng = loc.get("longitude")
        if lat is None or lng is None:
            continue
        slug = store.get("slug") or ""
        out.append(DeliveryRestaurant(
            platform="ubereats",
            id=store_id,
            name=store.get("title") or store.get("name", ""),
            lat=float(lat),
            lng=float(lng),
            url=f"https://www.ubereats.com/de/store/{slug}/{store_id}" if slug else f"https://www.ubereats.com/de/store/_/{store_id}",
            address=loc.get("address", ""),
        ))
        seen.add(store_id)
    return out
```

```python
import datetime as dt
import json
from dataclasses import dataclass, asdict
from pathlib import Path
from typing import Optional

from . import db as _db
from .adapters.base import DeliveryRestaurant
from .config import DB_PATH, DELIVERY_CACHE_TTL_HOURS


@dataclass
class CacheEntry:
    status: str
    results: list[DeliveryRestaurant]
    error: Optional[str]
    fetched_at: dt.datetime


def _now() -> dt.datetime:
    return dt.datetime.utcnow()


def read(address_norm: str, platform: str, db_path: Path = DB_PATH) -> Optional[CacheEntry]:
    cutoff = _now() - dt.timedelta(hours=DELIVERY_CACHE_TTL_HOURS)
    with _db.connect(db_path) as conn:
        row = conn.execute(
            "SELECT payload, status, error, fetched_at "
            "FROM delivery_cache WHERE address_norm=? AND platform=?",
            (address_norm, platform),
        ).fetchone()
    if row is None:
        return None
    fetched = dt.datetime.fromisoformat(row["fetched_at"])
    if fetched < cutoff:
        return None
    payload = json.loads(row["payload"])
    results = [DeliveryRestaurant(**r) for r in payload]
    return CacheEntry(
        status=row["status"],
        results=results,
        error=row["error"],
        fetched_at=fetched,
    )


def write(
    address_norm: str,
    platform: str,
    status: str,
    results: list[DeliveryRestaurant],
    error: Optional[str],
    db_path: Path = DB_PATH,
) -> None:
    payload = json.dumps([asdict(r) for r in results])
    with _db.connect(db_path) as conn:
        conn.execute(
            "INSERT OR REPLACE INTO delivery_cache "
            "(address_norm, platform, payload, status, error, fetched_at) "
            "VALUES (?,?,?,?,?,?)",
            (address_norm, platform, payload, status, error, _now().isoformat()),
        )


def invalidate(address_norm: str, db_path: Path = DB_PATH) -> int:
    with _db.connect(db_path) as conn:
        cur = conn.execute(
            "DELETE FROM delivery_cache WHERE address_norm=?", (address_norm,),
        )
        return cur.rowcount
```

```js
async refresh() {
  if (!this.address.trim()) return;
  this.loading = true; this.error = null;
  try {
    await fetch('/api/refresh', {
      method: 'POST',
      headers: {'Content-Type': 'application/json'},
      body: JSON.stringify({address: this.address}),
    });
    await this.check();
  } catch (e) {
    this.error = String(e.message || e);
  } finally {
    this.loading = false;
  }
```
